    68. 

  68.     private function update_list($path)

  78.         foreach ($raw_list as $file)

  79.         {

  80.             if ($file == '')

  83.             }

  85.             if ($name === false)
    86.

    87.

    88. 

  88.             }
    89. 

  89.             $this -> is_directory[] = (strtolower($file{0}) === 'd');

  90.             $this -> contents[] = $path . substr($name, 1);
    91. 

  91.         }

  92.         $this -> list_count = count($this -> contents);
    93.

    94. 

  94.     }

    96. 

  96.     /**

  97.      * @param string $local

  98.      * @param string $remote

  99.      */

  102.         if (!@ftp_get($this -> handle, $local, $remote, FTP_BINARY))

    121. 

  121.      * @param string $host

  122.      * @param int $port

  123.      * @param bool $passive

  129.     {

    131. 

  131.         if ($this -> handle === false)

  132.         {
    133. 

  133.             throw new ExceptionDisplay('Could not connect to FTP server.');

    138. 

  138.         }
    139. 

  139.         if ($passive && !@ftp_pasv($this -> handle, true))
    140. 

  140.         {
    141. 

  141.             throw new ExceptionDisplay('Could not set passive mode for FTP server.');
    142. 

  142.         }
    143. 

  143.         $this -> update_list($directory);
    144. 

  144.     }
