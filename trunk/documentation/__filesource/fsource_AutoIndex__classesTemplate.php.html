  33. /**
    34. 

  34.  * Reads the file contents, then parses comments and translated words.
    35. 

  35.  *
    36. 

  36.  * First step in parsing a template file. Used on all templates:
    37. 

  37.  * - global header
    38. 

  38.  * - global footer
    39. 

  39.  * - table header
    40. 

  40.  * - table footer
    41. 

  41.  * - each_file
    42. 

  42.  *
    43. 

  43.  * @author Justin Hagstrom <JustinHagstrom@yahoo.com>
    44. 

  44.  * @version 1.0.3 (February 02, 2005)
    45. 

  45.  * @package AutoIndex
    46. 

  46.  */
    47. 

  47. class Template
    48. 

  48. {
    49. 

  49.     /**
    50. 

  50.      * @var string The final output
    51. 

  51.      */
    52. 

  52.     protected $out;
    53. 

  53.     
    54. 

  54.     /**
    55. 

  55.      * @param array $m The array given by preg_replace_callback()
    56. 

  56.      * @return string Looks up $m[1] in word list and returns match
    57. 

  57.      */
    58. 

  58.     private static function callback_words($m)
    59. 

  59.     {
    60. 

  60.         global $words;
    61. 

  61.         return $words -> __get(strtolower($m[1]));
    62. 

  62.     }
    63. 

  63.     
    64. 

  64.     /**
    65. 

  65.      * @param array $m The array given by preg_replace_callback()
    66. 

  66.      * @return string The parsed template of filename $m[1]
    67. 

  67.      */
    68. 

  68.     private static function callback_include($m)
    69. 

  69.     {
    70. 

  70.         $temp = new Template($m[1]);
    71. 

  71.         return $temp -> __toString();
    72. 

  72.     }
    73. 

  73.     
    74. 

  74.     /**
    75. 

  75.      * @param array $m The array given by preg_replace_callback()
    76. 

  76.      * @return string The setting for the config value $m[1]
    77. 

  77.      */
    78. 

  78.     private static function callback_config($m)
    79. 

  79.     {
    80. 

  80.         global $config;
    81. 

  81.         return $config -> __get(strtolower($m[1]));
    82. 

  82.     }

    84. 

  84.     /**
    85. 

  85.      * Parses the text in $filename and sets the result to $out. We cannot
    86. 

  86.      * use ExceptionDisplay here if there is an error, since it uses the
    87. 

  87.      * template system.
    88. 

  88.      *
    89. 

  89.      * Steps to parse the template:
    90. 

  90.      * - remove comments
    91. 

  91.      * - replace {info} variables
    92. 

  92.      * - replace {words} strings
    93. 

  93.      * - replace {config} variables
    94. 

  94.      * - include other files when we see the {include} statement
    95. 

  95.      *
    96. 

  96.      * @param string $filename The name of the file to parse
    97. 

  97.      */
    98. 

  98.     public function __construct($filename)
    99. 

  99.     {
    100. 

  100.         global $config, $dir, $subdir;
    101. 

  101.         $full_filename = $config -> __get('template') . $filename;
    102. 

  102.         if (!@is_file($full_filename))
    103. 

  103.         {
    104. 

  104.             throw new ExceptionFatal('Template file <em>'
    105. 

  105.             . Url::html_output($full_filename) . '</em> cannot be found.');
    106. 

  106.         }
    107. 

  107.         
    108. 

  108.         //read raw file contents
    109. 

  109.         $contents = @file_get_contents($full_filename);
    110. 

  110.         if ($contents === false)
    111. 

  111.         {
    112. 

  112.             throw new ExceptionFatal('Template file <em>'
    113. 

  113.             . Url::html_output($full_filename) . '</em> could not be opened for reading.');
    114. 

  114.         }
    115. 

  115.         
    116. 

  116.         //remove comments
    117. 

  117.         $contents = preg_replace('#/\*.*?\*/#s', '', $contents);
    118. 

  118.         
    119. 

  119.         //replace info variables and word strings from language file
    120. 

  120.         $tr = array(
    121. 

  121.             '{info:dir}' => (isset($dir) ? Url::html_output($dir) : ''),
    122. 

  122.             '{info:subdir}' => (isset($subdir) ? Url::html_output($subdir) : ''),
    123. 

  123.             '{info:version}' => VERSION,
    124. 

  124.             '{info:page_time}' => round((microtime(true) - START_TIME) * 1000, 1));
    125. 

  125.         $contents = preg_replace_callback('/\{\s*words?\s*:\s*(.+)\s*\}/Ui',
    126. 

  126.             array('self', 'callback_words'), strtr($contents, $tr));
    127. 

  127.         
    128. 

  128.         //replace {config} variables
    129. 

  129.         $contents = preg_replace_callback('/\{\s*config\s*:\s*(.+)\s*\}/Ui',
    130. 

  130.             array('self', 'callback_config'), $contents);
    131. 

  131.  
    132. 

  132.         //parse includes
    133. 

  133.         $this -> out = preg_replace_callback('/\{\s*include\s*:\s*(.+)\s*\}/Ui',
    134. 

  134.             array('self', 'callback_include'), $contents);
    135. 

  135.     }
    136. 

  136.     
    137. 

  137.     /**
    138. 

  138.      * @return string The HTML text of the parsed template
    139. 

  139.      */
    140. 

  140.     public function __toString()
    141. 

  141.     {
    142. 

  142.         return $this -> out;
    143. 

  143.     }
