    71. 

  71.      * Checks input for hidden files/folders, and deals with ".."

  72.      *

  76.     private static function eval_dir($d)
    77. 

  77.     {
    78. 

  78.         $d = str_replace('\\', '/', $d);
    79. 

  79.         if ($d == '' || $d == '/')
    80. 

  80.         {
    81. 

  81.             return '';
    82. 

  82.         }
    83. 

  83.         $dirs = explode('/', $d);
    84. 

  84.         for ($i = 0; $i < count($dirs); $i++)
    85. 

  85.         {
    86. 

  86.             if (DirectoryList::is_hidden($dirs[$i], false))
    87. 

  87.             {
    88. 

  88.                 array_splice($dirs, $i, 1);
    89. 

  89.                 $i--;
    90. 

  90.             }
    91. 

  91.             else if (preg_match('/^\.\./', $dirs[$i])) //if it starts with two dots
    92. 

  92.             {
    93. 

  93.                 array_splice($dirs, $i-1, 2);
    94. 

  94.                 $i = -1;
    95. 

  95.             }
    96. 

  96.         }
    97. 

  97.         $new_dir = implode('/', $dirs);
    98. 

  98.         if ($new_dir == '' || $new_dir == '/')
    99. 

  99.         {
    100. 

  100.             return '';
    101. 

  101.         }
    102. 

  102.         if ($d{0} == '/' && $new_dir{0} != '/')
    103. 

  103.         {
    104. 

  104.             $new_dir = '/' . $new_dir;
    105. 

  105.         }
    106. 

  106.         if (preg_match('#/$#', $d) && !preg_match('#/$#', $new_dir))
    107. 

  107.         {
    108. 

  108.             $new_dir .= '/';
    109. 

  109.         }
    110. 

  110.         else if (DirectoryList::is_hidden(Item::get_basename($new_dir)))
    111. 

  111.         //it's a file, so make sure the file itself is not hidden
    112. 

  112.         {
    113. 

  113.             return DirItem::get_parent_dir($new_dir);
    114. 

  114.         }
    115. 

  115.         return $new_dir;
    116. 

  116.     }

  142.     public static function force_download($file_dl, $headers = true)

  143.     {
    144. 

  144.         if (!@is_file($file_dl))
    145. 

  145.         {
    146. 

  146.             header('HTTP/1.0 404 Not Found');

  151.         if (!($fn = @fopen($file_dl, 'rb')))

    155. 

  155.         }
    156. 

  156.         if ($headers)

  157.         {
    158. 

  158.             $outname = Item::get_basename($file_dl);

  159.             $size = @filesize($file_dl);

  160.             if ($size !== false)

  162.                 header('Content-Length: ' . $size);

  163.             }

  168.         global $speed;

  169.         while (true)

  170.         {

  172.             if ($temp === '')

  173.             {
    174. 

  174.                 break;
    175. 

  175.             }

  176.             echo $temp;
    177. 

  177.             flush();
    178. 

  178.             if (BANDWIDTH_LIMIT)
    179. 

  179.             {
    180. 

  180.                 sleep(1);
    181. 

  181.             }
    182. 

  182.         }

  184.     }
