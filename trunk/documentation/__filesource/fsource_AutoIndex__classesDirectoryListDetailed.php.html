  77.     private function set_path_nav()

  78.     {
    79. 

  79.         global $config, $subdir;

  80.         $exploded = explode('/', $subdir);
    81.

    84.

    89. 

  89.             {
    90. 

  90.                 $temp .= Url::translate_uri($exploded[$j]) . '/';
    91.

  97.     /**
    98. 

  98.      * Returns -1 if $a < $b or 1 if $a > $b
    99. 

  99.      *
    100. 

  100.      * @param Item $a 
    101. 

  101.      * @param Item $b

    104. 

  104.     private static function callback_sort(Item $a, Item $b)
    105. 

  105.     {
    106. 

  106.         if ($a -> __get('is_parent_dir'))
    107. 

  107.         {
    108. 

  108.             return -1;
    109. 

  109.         }
    110. 

  110.         if ($b -> __get('is_parent_dir'))
    111. 

  111.         {
    112. 

  112.             return 1;
    113. 

  113.         }
    114. 

  114.         $sort = strtolower($_SESSION['sort']);
    115. 

  115.         if ($sort === 'size')
    116. 

  116.         {
    117. 

  117.             $val = (($a -> __get('size') -> __get('bytes') <
    118. 

  118.                 $b -> __get('size') -> __get('bytes')) ? -1 : 1);
    119. 

  119.         }
    120. 

  120.         else
    121. 

  121.         {
    122. 

  122.             if (!$a -> is_set($sort))
    123. 

  123.             {
    124. 

  124.                 $_SESSION['sort'] = 'filename'; //so the "continue" link will work
    125. 

  125.                 throw new ExceptionDisplay('Invalid sort mode.');
    126. 

  126.             }
    127. 

  127.             if (is_string($a -> __get($sort)))
    128. 

  128.             {
    129. 

  129.                 $val = strnatcasecmp($a -> __get($sort), $b -> __get($sort));
    130. 

  130.             }
    131. 

  131.             else
    132. 

  132.             {
    133. 

  133.                 $val = (($a -> __get($sort) < $b -> __get($sort)) ? -1 : 1);
    134. 

  134.             }
    135. 

  135.         }
    136. 

  136.         return ((strtolower($_SESSION['sort_mode']) === 'd') ? -$val : $val);
    137. 

  137.     }

  171.                 if ($temp -> __get('is_parent_dir'))
    172. 

  172.                 {

  173.                     $dirs[] = $temp;

  174.                     $subtract_parent = true;
    175.

  176.                 else if ($temp -> __get('filename') !== false)

  177.                 {
    178. 

  178.                     $dirs[] = $temp;
    179. 

  179.                     if ($temp -> __get('size') -> __get('bytes') !== false)
    180. 

  180.                     {
    181. 

  181.                         $total_size += $temp -> __get('size') -> __get('bytes');
    182. 

  182.                     }
    183. 

  183.                 }
    184. 

  184.             }

  185.             else if (@is_file($path . $t))
    186. 

  186.             {
    187. 

  187.                 $temp = new FileItem($path, $t);

  188.                 if ($temp -> __get('filename') !== false)

  189.                 {

  190.                     $files[] = $temp;

  191.                     $this -> total_downloads += $temp -> __get('downloads');
    192. 

  192.                     $total_size += $temp -> __get('size') -> __get('bytes');
    193. 

  193.                 }
    194. 

  194.             }
    195. 

  195.         }
    196. 

  196.         self::sort_list($dirs);
    197.

  198.         $this -> contents = array_merge($dirs, $files);
    199. 

  199.         $this -> total_size = new Size($total_size);
    200. 

  200.         $this -> total_files = count($files);
    201. 

  201.         $this -> raw_total_folders = $this -> total_folders = count($dirs);
    202. 

  202.         if ($subtract_parent)
    203. 

  203.         {
    204. 

  204.             $this -> total_folders--;
    205. 

  205.         }
    206. 

  206.         $this -> path_nav = $this -> set_path_nav();
    207. 

  207.         
    208. 

  208.         //Paginate the files
    209. 

  209.         if (ENTRIES_PER_PAGE)
    210. 

  210.         {
    211. 

  211.             if ($page < 1)
    212. 

  212.             {
    213. 

  213.                 throw new ExceptionDisplay('Invalid page number.');
    214. 

  214.             }
    215. 

  215.             global $config;
    216. 

  216.             $num_per_page = $config -> __get('entries_per_page');
    217. 

  217.             if (($page - 1) * $num_per_page >= $this -> total_items())
    218. 

  218.             {
    219. 

  219.                 throw new ExceptionDisplay('Invalid page number.');
    220. 

  220.             }
    221. 

  221.             $this -> contents = array_slice($this -> contents, ($page - 1) * $num_per_page, $num_per_page);
    222. 

  222.         }
    223. 

  223.     }
