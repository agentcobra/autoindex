    75. 

  75.     private function callback_file($m)
    76. 

  76.     {
    77. 

  77.         global $words, $subdir;
    78. 

  78.         switch (strtolower($m[1]))
    79. 

  79.         {
    80. 

  80.             case 'tr_class':
    81. 

  81.             {
    82. 

  82.                 return (($this -> i % 2) ? 'dark_row' : 'light_row');
    83. 

  83.             }
    84. 

  84.             case 'filename':
    85. 

  85.             {
    86. 

  86.                 return Url::html_output($this -> temp_item -> __get('filename'));
    87. 

  87.             }
    88. 

  88.             case 'file_ext':
    89. 

  89.             {
    90. 

  90.                 return $this -> temp_item -> file_ext();
    91. 

  91.             }
    92. 

  92.             case 'size':
    93. 

  93.             {
    94. 

  94.                 return $this -> temp_item -> __get('size') -> formatted();
    95. 

  95.             }
    96. 

  96.             case 'bytes':
    97. 

  97.             {
    98. 

  98.                 return $this -> temp_item -> __get('size') -> __get('bytes');
    99. 

  99.             }
    100. 

  100.             case 'date':
    101. 

  101.             case 'time':
    102. 

  102.             case 'm_time':
    103. 

  103.             {
    104. 

  104.                 return $this -> temp_item -> format_m_time();
    105. 

  105.             }
    106. 

  106.             case 'a_time':
    107. 

  107.             {
    108. 

  108.                 return $this -> temp_item -> format_a_time();
    109. 

  109.             }
    110. 

  110.             case 'thumbnail':
    111. 

  111.             {
    112. 

  112.                 return $this -> temp_item -> __get('thumb_link');
    113. 

  113.             }
    114. 

  114.             case 'num_subfiles':
    115. 

  115.             {
    116. 

  116.                 return (($this -> temp_item instanceof DirItem 
    117. 

  117.                 && !$this -> temp_item -> __get('is_parent_dir')) ? $this -> temp_item -> num_subfiles() : '');
    118. 

  118.             }
    119. 

  119.             case 'delete_link':
    120. 

  120.             {
    121. 

  121.                 return (($this -> is_admin && !$this -> temp_item -> __get('is_parent_dir')) ?
    122. 

  122.                 ' [<a href="' . $_SERVER['PHP_SELF'] . '?action=delete&amp;dir=' . rawurlencode($subdir)
    123. 

  123.                 . '&amp;filename=' . rawurlencode($this -> temp_item -> __get('filename'))
    124. 

  124.                 . '" class="autoindex_small autoindex_a">' . $words -> __get('delete') . '</a>]' : '');
    125. 

  125.             }
    126. 

  126.             case 'rename_link':
    127. 

  127.             {
    128. 

  128.                 return (($this -> is_admin && !$this -> temp_item -> __get('is_parent_dir')) ?
    129. 

  129.                 ' [<a href="' . $_SERVER['PHP_SELF'] . '?action=rename&amp;dir=' . rawurlencode($subdir)
    130. 

  130.                 . '&amp;filename=' . rawurlencode($this -> temp_item -> __get('filename'))
    131. 

  131.                 . '" class="autoindex_small autoindex_a">' . $words -> __get('rename') . '</a>]' : '');
    132. 

  132.             }
    133. 

  133.             case 'edit_description_link':
    134. 

  134.             {
    135. 

  135.                 $slash = (($this -> temp_item instanceof DirItem) ? '/' : '');
    136. 

  136.                 return (($this -> is_mod && DESCRIPTION_FILE && !$this -> temp_item -> __get('is_parent_dir')) ?
    137. 

  137.                 ' [<a href="' . $_SERVER['PHP_SELF'] . '?action=edit_description&amp;dir='
    138. 

  138.                 . rawurlencode($subdir) . '&amp;filename='
    139. 

  139.                 . rawurlencode($this -> temp_item -> __get('filename')) . $slash
    140. 

  140.                 . '" class="autoindex_small autoindex_a">'
    141. 

  141.                 . $words -> __get('edit description') . '</a>]' : '');
    142. 

  142.             }
    143. 

  143.             case 'ftp_upload_link':
    144. 

  144.             {
    145. 

  145.                 if (!$this -> is_mod || !$this -> temp_item instanceof FileItem || !isset($_SESSION['ftp']))
    146. 

  146.                 {
    147. 

  147.                     return '';
    148. 

  148.                 }
    149. 

  149.                 return ' [<a href="' . $_SERVER['PHP_SELF'] . '?action=ftp&amp;dir='
    150. 

  150.                 . rawurlencode($subdir) . '&amp;filename=' . rawurlencode($this -> temp_item -> __get('filename'))
    151. 

  151.                 . '" class="autoindex_small autoindex_a">' . $words->__get('upload to ftp') . '</a>]';
    152. 

  152.             }
    153. 

  153.             default:
    154. 

  154.             {
    155. 

  155.                 return $this -> temp_item -> __get($m[1]);
    156. 

  156.             }
    157. 

  157.         }

  160.     /**
    161. 

  161.      * Either the HTML text is returned, or an empty string is returned,
    162. 

  162.      * depending on if the if-statement passed.
    163. 

  163.      *
    164. 

  164.      * @param array $m The array given by preg_replace_callback()

    167. 

  167.     private function callback_type($m)
    168. 

  168.     {
    169. 

  169.         switch (strtolower($m[1]))
    170. 

  170.         {
    171. 

  171.             case 'is_file': //file
    172. 

  172.             {
    173. 

  173.                 return (($this -> temp_item instanceof FileItem) ? $m[2] : '');
    174. 

  174.             }
    175. 

  175.             case 'is_dir': //folder or link to parent directory
    176. 

  176.             {
    177. 

  177.                 return (($this -> temp_item instanceof DirItem) ? $m[2] : '');
    178. 

  178.             }
    179. 

  179.             case 'is_real_dir': //folder
    180. 

  180.             {
    181. 

  181.                 return (($this -> temp_item instanceof DirItem
    182. 

  182.                 && !$this -> temp_item -> __get('is_parent_dir')) ? $m[2] : '');
    183. 

  183.             }
    184. 

  184.             case 'is_parent_dir': //link to parent directory
    185. 

  185.             {
    186. 

  186.                 return (($this -> temp_item instanceof DirItem
    187. 

  187.                 && $this -> temp_item -> __get('is_parent_dir')) ? $m[2] : '');
    188. 

  188.             }
    189. 

  189.             default:
    190. 

  190.             {
    191. 

  191.                 throw new ExceptionDisplay('Invalid file:if statement in <em>'
    192. 

  192.                 . Url::html_output(EACH_FILE) . '</em>');
    193. 

  193.             }
    194. 

  194.         }
    195. 

  195.     }

  197.     /**
    198. 

  198.      * Either the HTML text is returned or an empty string is returned,
    199. 

  199.      * depending on if temp_item is the ith file parsed.
    200. 

  200.      *
    201. 

  201.      * @param array $m The array given by preg_replace_callback()
    202. 

  202.      * @return string The result to insert into the HTML output
    203. 

  203.      */
    204. 

  204.     private function callback_do_every($m)
    205. 

  205.     {
    206. 

  206.         $num = $this -> i + 1;
    207. 

  207.         return (($num % (int)$m[1] === 0 && $this -> length !== $num) ? $m[2] : '');
    208. 

  208.     }

  211.     /**
    212. 

  212.      * Parses info for each file in the directory. Order of elements to
    213. 

  213.      * replace is:
    214. 

  214.      * - file:if
    215. 

  215.      * - do_every
    216. 

  216.      * - file
    217. 

  217.      *
    218. 

  218.      * @param string $filename The name of the file to parse
    219. 

  219.      * @param DirectoryListDetailed $list 
    220. 

  220.      */
    221. 

  221.     public function __construct($filename, DirectoryListDetailed $list)
    222. 

  222.     {
    223. 

  223.         parent::__construct($filename, $list);
    224. 

  224.         global $you;
    225. 

  225.         $this -> is_admin = ($you -> level >= ADMIN);
    226. 

  226.         $this -> is_mod = ($you -> level >= MODERATOR);
    227. 

  227.         $final_file_line = '';
    228. 

  228.         $this -> length = (int)$list -> __get('list_count');
    229. 

  229.         foreach ($list as $i => $item)
    230. 

  230.         {
    231. 

  231.             $this -> i = (int)$i;
    232. 

  232.             $this -> temp_item = $item;
    233. 

  233.             $temp_line = preg_replace_callback('/\{\s*file\s*:\s*if\s*:\s*(\w+)\s*\}(.*)\{\s*end\s*if\s*\}/Uis',
    234. 

  234.                 array($this, 'callback_type'), $this -> out);
    235. 

  235.             $temp_line = preg_replace_callback('/\{\s*do_every\s*:\s*(\d+)\s*\}(.*)\{\s*end\s*do_every\s*\}/Uis',
    236. 

  236.                 array($this, 'callback_do_every'), $temp_line);
    237. 

  237.             $final_file_line .= preg_replace_callback('/\{\s*file\s*:\s*(\w+)\s*\}/Ui',
    238. 

  238.                 array($this, 'callback_file'), $temp_line);
    239. 

  239.         }
    240. 

  240.         $this -> out = $final_file_line;
    241. 

  241.     }
    242. 

  242. }
