  48.     /**
    49. 

  49.      * @return string The HTML text that makes up the search box

    51. 

  51.     public static function search_box()
    52. 

  52.     {
    53. 

  53.         global $words, $subdir;
    54. 

  54.         $search = (isset($_GET['search']) ? Url::html_output($_GET['search']) : '');
    55. 

  55.         $mode = (isset($_GET['search_mode']) ? self::clean_mode($_GET['search_mode']) : 'f');
    56. 

  56.         $modes = array('files' => 'f', 'folders' => 'd', 'both' => 'fd');

    64. 

  64.             $out .= '<option value="' . $m . '"' . $sel . '>'
    65. 

  65.             . $words -> __get($word) . '</option>';
    66. 

  66.         }
    67. 

  67.         $out .= '</select><input type="submit" class="button" value="'

  68.         . $words -> __get('search') . '" /></p></form>';

  70.     }

  72.     /**
    73. 

  73.      * @param string $filename 
    74. 

  74.      * @param string $string

  75.      * @return bool True if string matches filename

    77. 

  77.     private static function match(&$filename, &$string)
    78. 

  78.     {
    79. 

  79.         if (preg_match_all('/(?<=")[^"]+(?=")|[^ "]+/', $string, $matches))
    80. 

  80.         {
    81. 

  81.             foreach ($matches[0] as $w)
    82. 

  82.             {
    83. 

  83.                 if (stripos($filename, $w) !== false)
    84. 

  84.                 {
    85. 

  85.                     return true;
    86. 

  86.                 }
    87. 

  87.             }
    88. 

  88.         }
    89. 

  89.         return false;
    90. 

  90.     }

  93.      * Merges $obj into $this.

  95.      * @param Search $obj

  97.     private function merge(Search $obj)

  99.         $this -> total_folders += $obj -> __get('total_folders');

  100.         $this -> total_files += $obj -> __get('total_files');

  103.         $this -> matches = array_merge($this -> matches, $obj -> __get('contents'));

  106.     /**
    107. 

  107.      * Returns a string with all characters except 'd' and 'f' stripped.

  108.      * Either 'd' 'f' 'df' will be returned, defaults to 'f'

  109.      *
    110. 

  110.      * @param string $mode 
    111. 

  111.      * @return string

  112.      */
    113. 

  113.     private static function clean_mode($mode)
    114. 

  114.     {
    115. 

  115.         $str = '';
    116. 

  116.         if (stripos($mode, 'f') !== false)
    117. 

  117.         {
    118. 

  118.             $str .= 'f';
    119. 

  119.         }
    120. 

  120.         if (stripos($mode, 'd') !== false)
    121. 

  121.         {
    122. 

  122.             $str .= 'd';
    123. 

  123.         }
    124. 

  124.         else if ($str == '')
    125. 

  125.         {
    126. 

  126.             $str = 'f';
    127. 

  127.         }
    128. 

  128.         return $str;
    129. 

  129.     }

  131.     /**
    132. 

  132.      * @param string $query String to search for
    133. 

  133.      * @param string $dir The folder to search (recursive)
    134. 

  134.      * @param string $mode Should be f (files), d (directories), or fd (both)
    135. 

  135.      */
    136. 

  136.     public function __construct($query, $dir, $mode)
    137. 

  137.     {
    138. 

  138.         if (strlen($query) < 2 || strlen($query) > 20)
    139. 

  139.         {
    140. 

  140.             throw new ExceptionDisplay('Search query is either too long or too short.');
    141. 

  141.         }
    142. 

  142.         $mode = self::clean_mode($mode);
    143. 

  143.         $dir = Item::make_sure_slash($dir);
    144. 

  144.         DirectoryList::__construct($dir);
    145. 

  145.         $this -> matches = array();
    146. 

  146.         $this -> total_size = new Size(0);
    147. 

  147.         $this -> total_downloads = $this -> total_folders = $this -> total_files = 0;
    148. 

  148.         foreach ($this as $item)
    149. 

  149.         {
    150. 

  150.             if ($item == '..')
    151. 

  151.             {
    152. 

  152.                 continue;
    153. 

  153.             }
    154. 

  154.             if (@is_dir($dir . $item))
    155. 

  155.             {
    156. 

  156.                 if (stripos($mode, 'd') !== false && self::match($item, $query))
    157. 

  157.                 {
    158. 

  158.                     $temp = new DirItem($dir, $item);
    159. 

  159.                     $this -> matches[] = $temp;
    160. 

  160.                     if ($temp -> __get('size') -> __get('bytes') !== false)
    161. 

  161.                     {
    162. 

  162.                         $this -> total_size -> add_size($temp -> __get('size'));
    163. 

  163.                     }
    164. 

  164.                     $this -> total_folders++;
    165. 

  165.                 }
    166. 

  166.                 $sub_search = new Search($query, $dir . $item, $mode);
    167. 

  167.                 $this -> merge($sub_search);
    168. 

  168.             }
    169. 

  169.             else if (stripos($mode, 'f') !== false && self::match($item, $query))
    170. 

  170.             {
    171. 

  171.                 $temp = new FileItem($dir, $item);
    172. 

  172.                 $this -> matches[] = $temp;
    173. 

  173.                 $this -> total_size -> add_size($temp -> __get('size'));
    174. 

  174.                 $this -> total_downloads += $temp -> __get('downloads');
    175. 

  175.                 $this -> total_files++;
    176. 

  176.             }
    177. 

  177.         }
    178. 

  178.         global $words, $config, $subdir;
    179. 

  179.         $link = ' <a class="autoindex_a" href="' . $_SERVER['PHP_SELF']
    180. 

  180.         . '?dir=' . Url::translate_uri($subdir) . '">'
    181. 

  181.         . Url::html_output($dir) . '</a> ';
    182. 

  182.         $this -> path_nav = $words -> __get('search results for')
    183. 

  183.         . $link . $words -> __get('and its subdirectories');
    184. 

  184.         $this -> contents = $this -> matches;
    185. 

  185.         unset($this -> matches);
    186. 

  186.     }
    187. 

  187. }
