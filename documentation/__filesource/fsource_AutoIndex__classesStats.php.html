  84.     private static function get_change_color($num)
    85. 

  85.     {
    86. 

  86.         if ($num > 0)

  87.         {

  88.             return '<span style="color: #00FF00;">+';

  104.     private static function add_num_to_array($num, &$array)
    105. 

  105.     {
    106. 

  106.         isset($array[$num]) ? $array[$num]++ : $array[$num] = 1;

  109.     /**
    110. 

  110.      * Reads the log file, and sets the member variables after doing

    113. 

  113.     public function __construct()
    114. 

  114.     {
    115. 

  115.         $extensions = $dates = $unique_hits = $countries = array();
    116. 

  116.         $total_hits = 0;
    117. 

  117.         global $config;
    118. 

  118.         $log_file = $config -> __get('log_file');
    119. 

  119.         $base_dir = $config -> __get('base_dir');
    120. 

  120.         $h = @fopen($log_file, 'rb');
    121. 

  121.         if ($h === false)
    122. 

  122.         {
    123. 

  123.             throw new ExceptionDisplay("Cannot open log file: <em>$log_file</em>");
    124. 

  124.         }
    125. 

  125.         while (!feof($h))
    126. 

  126.         {
    127. 

  127.             $entries = explode("\t", rtrim(fgets($h, 1024), "\r\n"));
    128. 

  128.             if (count($entries) === 7)
    129. 

  129.             {
    130. 

  130.                 //find the number of unique visits
    131. 

  131.                 if ($entries[5] == $base_dir)
    132. 

  132.                 {
    133. 

  133.                     $total_hits++;
    134. 

  134.                     if (!in_array($entries[3], $unique_hits))
    135. 

  135.                     {
    136. 

  136.                         $unique_hits[] = Url::html_output($entries[3]);
    137. 

  137.                     }
    138. 

  138.     
    139. 

  139.                     //find country codes by hostnames
    140. 

  140.                     $cc = FileItem::ext($entries[3]);
    141. 

  141.                     if (preg_match('/^[a-z]+$/i', $cc))
    142. 

  142.                     {
    143. 

  143.                         self::add_num_to_array($cc, $countries);
    144. 

  144.                     }
    145. 

  145.     
    146. 

  146.                     //find the dates of the visits
    147. 

  147.                     self::add_num_to_array($entries[0], $dates);
    148. 

  148.                 }
    149. 

  149.     
    150. 

  150.                 //find file extensions
    151. 

  151.                 $ext = FileItem::ext($entries[6]);
    152. 

  152.                 if (preg_match('/^[\w-]+$/', $ext))
    153. 

  153.                 {
    154. 

  154.                     self::add_num_to_array($ext, $extensions);
    155. 

  155.                 }
    156. 

  156.             }
    157. 

  157.         }
    158. 

  158.         fclose($h);
    159. 

  159.         $this -> num_days = count($dates);
    160. 

  160.         $this -> avg = round($total_hits / $this -> num_days);
    161. 

  161.         $this -> extensions = $extensions;
    162. 

  162.         $this -> dates = $dates;
    163. 

  163.         $this -> unique_hits = $unique_hits;
    164. 

  164.         $this -> countries = $countries;
    165. 

  165.         $this -> total_hits = $total_hits;
    166. 

  166.     }
