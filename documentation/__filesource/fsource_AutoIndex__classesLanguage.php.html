  52.     /**
    53. 

  53.      * Returns a list of all files in $path that match the filename format
    54. 

  54.      * of language files.
    55. 

  55.      *
    56. 

  56.      * There are two valid formats for the filename of a language file. The
    57. 

  57.      * standard way is the language code then the .txt extension. You can
    58. 

  58.      * also use the language code followed by an underscore then the
    59. 

  59.      * country code. The second format would be used for dialects of
    60. 

  60.      * languages. For example pt.txt would be Portuguese, and pt_BR.txt
    61. 

  61.      * would be Brazilian Portuguese. The filenames are case insensitive.
    62. 

  62.      *
    63. 

  63.      * @param string $path The directory to read from
    64. 

  64.      * @return array The list of valid language files (based on filename)
    65. 

  65.      */
    66. 

  66.     public static function get_all_langs($path)
    67. 

  67.     {
    68. 

  68.         if (($hndl = @opendir($path)) === false)
    69. 

  69.         {
    70. 

  70.             return false;
    71. 

  71.         }
    72. 

  72.         $list = array();
    73. 

  73.         while (($file = readdir($hndl)) !== false)
    74. 

  74.         {
    75. 

  75.             if (@is_file($path . $file) && preg_match('/^[a-z]{2}(_[a-z]{2})?'
    76. 

  76.             . preg_quote(LANGUAGE_FILE_EXT, '/') . '$/i', $file))
    77. 

  77.             {
    78. 

  78.                 $list[] = $file;
    79. 

  79.             }
    80. 

  80.         }
    81. 

  81.         closedir($hndl);
    82. 

  82.         for ($i = 0; $i < count($list); $i++)
    83. 

  83.         {
    84. 

  84.             //remove the file extention from each language code
    85. 

  85.             $list[$i] = substr($list[$i], 0, -strlen(LANGUAGE_FILE_EXT));
    86. 

  86.         }
    87. 

  87.         return $list;
    88. 

  88.     }
    89. 

  89.     
    90. 

  90.     /**
    91. 

  91.      * @return string The code for the language to load
    92. 

  92.      *
    93. 

  93.      *  First tries to use the default of the user's browser, and then tries
    94. 

  94.      *  the default in the config file.
    95. 

  95.      */
    96. 

  96.     private function get_current_lang()
    97. 

  97.     {
    98. 

  98.         //try to detect the default language of the user's browser
    99. 

  99.         if (isset($_SERVER['HTTP_ACCEPT_LANGUAGE']))
    100. 

  100.         //e.g. "en-us,en;q=0.5"
    101. 

  101.         {
    102. 

  102.             $available_langs = self::get_all_langs(PATH_TO_LANGUAGES);
    103. 

  103.             if ($available_langs !== false)
    104. 

  104.             {
    105. 

  105.                 $pref = array(); //user's preferred languages
    106. 

  106.                 foreach (split(',', strtolower($_SERVER['HTTP_ACCEPT_LANGUAGE'])) as $lang)
    107. 

  107.                 {
    108. 

  108.                     $lang_array = split(';q=', trim($lang));
    109. 

  109.                     $q = (isset($lang_array[1]) ? trim($lang_array[1]) : 1); //preference value
    110. 

  110.                     $pref[trim($lang_array[0])] = (float)$q;
    111. 

  111.                 }
    112. 

  112.                 arsort($pref);
    113. 

  113.                 //find the first match that is available:
    114. 

  114.                 foreach ($pref as $lang => $q)
    115. 

  115.                 {
    116. 

  116.                     if (in_array($lang, $available_langs))
    117. 

  117.                     {
    118. 

  118.                         return $lang;
    119. 

  119.                     }
    120. 

  120.                 }
    121. 

  121.             }
    122. 

  122.         }
    123. 

  123.         //the browser has no preferences set, so use the config's default
    124. 

  124.         global $config;
    125. 

  125.         return $config -> __get('language');
    126. 

  126.     }
    127. 

  127.     
    128. 

  128.     /**
    129. 

  129.      * Creates a new language object. First tries to use the default of
    130. 

  130.      * the user's browser, and then tries the default in the config file.
    131. 

  131.      */
    132. 

  132.     public function __construct()
    133. 

  133.     {
    134. 

  134.         $lang_file = PATH_TO_LANGUAGES . $this -> get_current_lang()
    135. 

  135.             . LANGUAGE_FILE_EXT;
    136. 

  136.         if (!@is_readable($lang_file))
    137. 

  137.         {
    138. 

  138.             throw new ExceptionFatal('Cannot read from language file: <em>' 
    139. 

  139.             . Url::html_output($lang_file) . '</em>');
    140. 

  140.         }
    141. 

  141.         //load the file as a tab-separated object
    142. 

  142.         $this -> translation_data = new ConfigData($lang_file);
    143. 

  143.     }
    144. 

  144.     
    145. 

  145.     /**
    146. 

  146.      * @param string $name The word to look for
    147. 

  147.      * @return bool True if $name is set in the translation file
    148. 

  148.      */
    149. 

  149.     public function is_set($name)
    150. 

  150.     {
    151. 

  151.         return $translation_data -> is_set($name);
    152. 

  152.     }    
    153. 

  153.     
    154. 

  154.     /**
    155. 

  155.      * @param string $var The key to look for (the keyword)
    156. 

  156.      * @return string The value $name points to (its translation)
    157. 

  157.      */
    158. 

  158.     public function __get($var)
    159. 

  159.     {
    160. 

  160.         if ($this -> translation_data -> is_set($var))
    161. 

  161.         {
    162. 

  162.             return $this -> translation_data -> __get($var);
    163. 

  163.         }
    164. 

  164.         throw new ExceptionDisplay('Variable <em>' . Url::html_output($var)
    165. 

  165.         . '</em> not set in Language file.');
    166. 

  166.     }
