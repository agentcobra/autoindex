  116.     private static function to64($value, $count)

  117.     {

  118.         static $root = './0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz';

  128.     /**
    129. 

  129.      * Implementation of Apache's Custom MD5 Crypt.
    130. 

  130.      *
    131. 

  131.      * @param string $plain The plaintext password
    132. 

  132.      * @param string $salt The salt
    133. 

  133.      * @return string The hashed password
    134. 

  134.      */
    135. 

  135.     private static function md5_crypt($plain, $salt)
    136. 

  136.     {
    137. 

  137.         $length = strlen($plain);
    138. 

  138.         $context = $plain . '$apr1$' . $salt;
    139. 

  139.         $binary = self::hex2bin(md5($plain . $salt . $plain));
    140. 

  140.         for ($i = $length; $i > 0; $i -= 16)
    141. 

  141.         {
    142. 

  142.             $context .= substr($binary, 0, min(16, $i));
    143. 

  143.         }
    144. 

  144.         for ( $i = $length; $i > 0; $i >>= 1)
    145. 

  145.         {
    146. 

  146.             $context .= ($i & 1) ? chr(0) : $plain[0];
    147. 

  147.         }
    148. 

  148.         $binary = self::hex2bin(md5($context));
    149. 

  149.         for ($i = 0; $i < 1000; $i++) 
    150. 

  150.         {
    151. 

  151.             $new = ($i & 1) ? $plain : substr($binary, 0, 16);
    152. 

  152.             if ($i % 3)
    153. 

  153.             {
    154. 

  154.                 $new .= $salt;
    155. 

  155.             }
    156. 

  156.             if ($i % 7)
    157. 

  157.             {
    158. 

  158.                 $new .= $plain;
    159. 

  159.             }
    160. 

  160.             $new .= (($i & 1) ? substr($binary, 0, 16) : $plain);
    161. 

  161.             $binary = self::hex2bin(md5($new));
    162. 

  162.         }
    163. 

  163.         $p = array();
    164. 

  164.         for ($i = 0; $i < 5; $i++)
    165. 

  165.         {
    166. 

  166.             $k = $i + 6;
    167. 

  167.             $j = $i + 12;
    168. 

  168.             if ($j == 16)
    169. 

  169.             {
    170. 

  170.                 $j = 5;
    171. 

  171.             }
    172. 

  172.             $p[] = self::to64(
    173. 

  173.                 (ord($binary[$i]) << 16) |
    174. 

  174.                 (ord($binary[$k]) << 8) |
    175. 

  175.                 (ord($binary[$j])), 5
    176. 

  176.                 );
    177. 

  177.         }
    178. 

  178.         return '$apr1$' . $salt . '$' . implode($p) . self::to64(ord($binary[11]), 3);
    179. 

  179.     }

  198.      * Checks if AuthName and AuthUserFile are set, and then prompts for a

    201. 

  201.     private function check_auth()
    202. 

  202.     {
    203. 

  203.         if ($this -> auth_user_file == '')
    204. 

  204.         {
    205. 

  205.             return;
    206. 

  206.         }
    207. 

  207.         if ($this -> auth_name == '')
    208. 

  208.         {
    209. 

  209.             $this -> auth_name = '"Directory access restricted by AutoIndex"';
    210. 

  210.         }
    211. 

  211.         $validated = false;
    212. 

  212.         if (isset($_SERVER['PHP_AUTH_USER'], $_SERVER['PHP_AUTH_PW']))
    213. 

  213.         {
    214. 

  214.             $file = @file($this -> auth_user_file);
    215. 

  215.             if ($file === false)
    216. 

  216.             {
    217. 

  217.                 $_GET['dir'] = '';
    218. 

  218.                 throw new ExceptionDisplay('Cannot open .htpasswd file.
    219. 

  219.                 <br /><em>' . htmlentities($this -> auth_user_file) . '</em>');
    220. 

  220.             }
    221. 

  221.             if ($this -> auth_required_users === array() || DirectoryList::match_in_array($_SERVER['PHP_AUTH_USER'], $this -> auth_required_users))
    222. 

  222.             {
    223. 

  223.                 foreach ($file as $account)
    224. 

  224.                 {
    225. 

  225.                     $parts = explode(':', trim($account));
    226. 

  226.                     if (count($parts) < 2 || $_SERVER['PHP_AUTH_USER'] != $parts[0])
    227. 

  227.                     {
    228. 

  228.                         continue;
    229. 

  229.                     }
    230. 

  230.                     if (isset($parts[2]))
    231. 

  231.                     //MD5 hash format with realm
    232. 

  232.                     {
    233. 

  233.                         $parts[1] = $parts[2];
    234. 

  234.                     }
    235. 

  235.                     switch (strlen($parts[1]))
    236. 

  236.                     {
    237. 

  237.                         case 13:
    238. 

  238.                         //Crypt hash format
    239. 

  239.                         {
    240. 

  240.                             $validated = (crypt($_SERVER['PHP_AUTH_PW'], substr($parts[1], 0, 2)) == $parts[1]);
    241. 

  241.                             break 2;
    242. 

  242.                         }
    243. 

  243.                         case 32:
    244. 

  244.                         //MD5 hash format
    245. 

  245.                         {
    246. 

  246.                             $validated = (md5($_SERVER['PHP_AUTH_PW']) == $parts[1]);
    247. 

  247.                             break 2;
    248. 

  248.                         }
    249. 

  249.                         case 37:
    250. 

  250.                         //Apache's MD5 Crypt hash format
    251. 

  251.                         {
    252. 

  252.                             $salt = explode('$', $parts[1]);
    253. 

  253.                             $validated = (self::md5_crypt($_SERVER['PHP_AUTH_PW'], $salt[2]) == $parts[1]);
    254. 

  254.                             break 2;
    255. 

  255.                         }
    256. 

  256.                         case 40:
    257. 

  257.                         //SHA-1 hash format
    258. 

  258.                         {
    259. 

  259.                             $validated = (sha1($_SERVER['PHP_AUTH_PW']) == $parts[1]);
    260. 

  260.                             break 2;
    261. 

  261.                         }
    262. 

  262.                     }
    263. 

  263.                 }
    264. 

  264.             }
    265. 

  265.         }
    266. 

  266.         if (!$validated)
    267. 

  267.         {
    268. 

  268.             header('WWW-Authenticate: Basic realm=' . $this -> auth_name);
    269. 

  269.             header('HTTP/1.0 401 Authorization Required');
    270. 

  270.             $_GET['dir'] = '';
    271. 

  271.             throw new ExceptionDisplay('A username and password are required to access this directory.');
    272. 

  272.         }
    273. 

  273.     }

  298.     /**
    299. 

  299.      * @param string $file The .htaccess file (name and path) to parse
    300.

    301. 

  301.     private function parse($file)
    302. 

  302.     {
    303. 

  303.         $data = @file($file);
    304. 

  304.         if ($data === false)
    305. 

  305.         {
    306. 

  306.             return;
    307. 

  307.         }
    308. 

  308.         $conditional_directory = '';
    309. 

  309.         $other_conditional = false;
    310. 

  310.         foreach ($data as $line)
    311. 

  311.         {
    312. 

  312.             $line = trim($line);
    313. 

  313.             if ($line == '')
    314. 

  314.             {
    315. 

  315.                 continue;
    316. 

  316.             }
    317. 

  317.             if ($line{0} == '<')
    318. 

  318.             {
    319. 

  319.                 if (preg_match('#^</\s*directory.*?>#i', $line))
    320. 

  320.                 {
    321. 

  321.                     $conditional_directory = '';
    322. 

  322.                 }
    323. 

  323.                 else if (preg_match('#^<\s*directory\s+\"?(.+?)\"?\s*>#i', $line, $matches))
    324. 

  324.                 {
    325. 

  325.                     $conditional_directory = $matches[1];
    326. 

  326.                 }
    327. 

  327.                 else if (preg_match('#^</?\s*limit.*?>#i', $line))
    328. 

  328.                 {
    329. 

  329.                     //ignore <Limit> tags
    330. 

  330.                     continue;
    331. 

  331.                 }
    332. 

  332.                 else if (preg_match('#^</\s*ifdefine.*?>#i', $line))
    333. 

  333.                 {
    334. 

  334.                     $conditional_defined = '';
    335. 

  335.                 }
    336. 

  336.                 else if (preg_match('#^<\s*ifdefine\s+(.+?)\s*>#i', $line, $matches))
    337. 

  337.                 {
    338. 

  338.                     $conditional_defined = $matches[1];
    339. 

  339.                 }
    340. 

  340.                 else if (isset($line{1}))
    341. 

  341.                 {
    342. 

  342.                     $other_conditional = ($line{1} != '/');
    343. 

  343.                 }
    344. 

  344.                 continue;
    345. 

  345.             }
    346. 

  346.             global $dir;
    347. 

  347.             if ($other_conditional || $conditional_directory != '' && !self::matches($conditional_directory, $dir))
    348. 

  348.             //deal with <Directory> or an unknown < > tag
    349. 

  349.             {
    350. 

  350.                 continue;
    351. 

  351.             }
    352. 

  352.             if ($conditional_defined != '')
    353. 

  353.             //deal with <IfDefine>
    354. 

  354.             {
    355. 

  355.                 $conditional_defined = strtoupper($conditional_defined);
    356. 

  356.                 if ($conditional_defined{0} === '!')
    357. 

  357.                 {
    358. 

  358.                     $conditional_defined = substr($conditional_defined, 1);
    359. 

  359.                     if (defined($conditional_defined) && constant($conditional_defined))
    360. 

  360.                     {
    361. 

  361.                         continue;
    362. 

  362.                     }
    363. 

  363.                 }
    364. 

  364.                 else if (!defined($conditional_defined) || !constant($conditional_defined))
    365. 

  365.                 {
    366. 

  366.                     continue;
    367. 

  367.                 }
    368. 

  368.             }
    369. 

  369.             $parts = preg_split('#\s#', $line, -1, PREG_SPLIT_NO_EMPTY);
    370. 

  370.             switch (strtolower($parts[0]))
    371. 

  371.             {
    372. 

  372.                 case 'indexignore':
    373. 

  373.                 {
    374. 

  374.                     global $hidden_files;
    375. 

  375.                     for ($i = 1; $i < count($parts); $i++)
    376. 

  376.                     {
    377. 

  377.                         $hidden_files[] = $parts[$i];
    378. 

  378.                     }
    379. 

  379.                     break;
    380. 

  380.                 }
    381. 

  381.                 case 'include':
    382. 

  382.                 {
    383. 

  383.                     if (isset($parts[1]) && @is_file($parts[1]) && @is_readable($parts[1]))
    384. 

  384.                     {
    385. 

  385.                         self::parse($parts[1]);
    386. 

  386.                     }
    387. 

  387.                     break;
    388. 

  388.                 }
    389. 

  389.                 case 'allow':
    390. 

  390.                 {
    391. 

  391.                     if (isset($parts[1]) && strtolower($parts[1]) === 'from')
    392. 

  392.                     {
    393. 

  393.                         for ($i = 2; $i < count($parts); $i++)
    394. 

  394.                         {
    395. 

  395.                             foreach (explode(',', $parts[$i]) as $ip)
    396. 

  396.                             {
    397. 

  397.                                 if (strtolower($ip) === 'all')
    398. 

  398.                                 {
    399. 

  399.                                     $this -> allow_list = array('*');
    400. 

  400.                                 }
    401. 

  401.                                 else
    402. 

  402.                                 {
    403. 

  403.                                     $this -> allow_list[] = $ip;
    404. 

  404.                                 }
    405. 

  405.                             }
    406. 

  406.                         }
    407. 

  407.                     }
    408. 

  408.                     break;
    409. 

  409.                 }
    410. 

  410.                 case 'deny':
    411. 

  411.                 {
    412. 

  412.                     if (isset($parts[1]) && strtolower($parts[1]) === 'from')
    413. 

  413.                     {
    414. 

  414.                         for ($i = 2; $i < count($parts); $i++)
    415. 

  415.                         {
    416. 

  416.                             foreach (explode(',', $parts[$i]) as $ip)
    417. 

  417.                             {
    418. 

  418.                                 if (strtolower($ip) === 'all')
    419. 

  419.                                 {
    420. 

  420.                                     $this -> deny_list = array('*');
    421. 

  421.                                 }
    422. 

  422.                                 else
    423. 

  423.                                 {
    424. 

  424.                                     $this -> deny_list[] = $ip;
    425. 

  425.                                 }
    426. 

  426.                             }
    427. 

  427.                         }
    428. 

  428.                     }
    429. 

  429.                     break;
    430. 

  430.                 }
    431. 

  431.                 case 'adddescription':
    432. 

  432.                 {
    433. 

  433.                     global $descriptions;
    434. 

  434.                     if (!isset($descriptions))
    435. 

  435.                     {
    436. 

  436.                         $descriptions = new ConfigData(false);
    437. 

  437.                     }
    438. 

  438.                     for ($i = 1; isset($parts[$i], $parts[$i+1]); $i += 2)
    439. 

  439.                     {
    440. 

  440.                         $descriptions -> set($parts[$i], $parts[$i+1]);
    441. 

  441.                     }
    442. 

  442.                     break;
    443. 

  443.                 }
    444. 

  444.                 case 'authuserfile':
    445. 

  445.                 {
    446. 

  446.                     if (isset($parts[1]))
    447. 

  447.                     {
    448. 

  448.                         $this -> auth_user_file = str_replace('"', '', implode(' ', array_slice($parts, 1)));
    449. 

  449.                     }
    450. 

  450.                     break;
    451. 

  451.                 }
    452. 

  452.                 case 'authname':
    453. 

  453.                 {
    454. 

  454.                     if (isset($parts[1]))
    455. 

  455.                     {
    456. 

  456.                         $this -> auth_name = implode(' ', array_slice($parts, 1));
    457. 

  457.                     }
    458. 

  458.                     break;
    459. 

  459.                 }
    460. 

  460.                 case 'order':
    461. 

  461.                 {
    462. 

  462.                     if (isset($parts[1]) && (strtolower($parts[1]) === 'allow,deny' || strtolower($parts[1]) === 'mutual-failure'))
    463. 

  463.                     {
    464. 

  464.                         $this -> order = 'allow,deny';
    465. 

  465.                     }
    466. 

  466.                 }
    467. 

  467.                 case 'require':
    468. 

  468.                 {
    469. 

  469.                     if (isset($parts[1]) && strtolower($parts[1]) === 'user')
