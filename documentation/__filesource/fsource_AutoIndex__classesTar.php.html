  33. /**
    34. 

  34.  * Creates and outputs a tar archive given an array of filenames.
    35. 

  35.  *
    36. 

  36.  * @author Justin Hagstrom <JustinHagstrom@yahoo.com>
    37.

    39. 

  39.  */
    40. 

  40. class Tar
    41. 

  41. {
    42. 

  42.     /**
    43. 

  43.      * @var int Length of directory path to cut off from start
    44. 

  44.      */
    45. 

  45.     private $base_dir_length;
    46. 

  46.     
    47. 

  47.     /**
    48. 

  48.      * @var string Added in the filepath inside the tar archive
    49. 

  49.      */
    50. 

  50.     private $prepend_path;
    51. 

  51.     
    52. 

  52.     /**
    53. 

  53.      * @param string $data 
    54. 

  54.      * @return int The checksum of $data
    55. 

  55.      */
    56. 

  56.     private static function checksum(&$data)
    57. 

  57.     {
    58. 

  58.         $unsigned_chksum = 0;
    59. 

  59.         for ($i = 0; $i < 512; $i++)
    60. 

  60.         {
    61. 

  61.             $unsigned_chksum += ord($data{$i});
    62. 

  62.         }
    63. 

  63.         for ($i = 148; $i < 156; $i++)
    64. 

  64.         {
    65. 

  65.             $unsigned_chksum -= ord($data{$i});
    66. 

  66.         }
    67. 

  67.         return $unsigned_chksum + 256;
    68. 

  68.     }
    69. 

  69.     
    70. 

  70.     /**
    71. 

  71.      * @param string $name The file or folder name
    72. 

  72.      * @param int $size The size of the file (0 for directories)
    73. 

  73.      * @param bool $is_dir True if folder, false if file
    74. 

  74.      */
    75. 

  75.     private function create_header($name, $size = 0, $is_dir = true)
    76. 

  76.     {
    77. 

  77.         $header = str_pad($this -> prepend_path . substr($name, $this -> base_dir_length), 100, "\0") //filename
    78. 

  78.         . str_pad('755', 7, '0', STR_PAD_LEFT) . "\0" //permissions
    79. 

  79.         . '0000000' . "\0" //uid
    80. 

  80.         . '0000000' . "\0" //gid
    81. 

  81.         . str_pad(decoct($size), 11, '0', STR_PAD_LEFT) . "\0" //size
    82. 

  82.         . str_pad(decoct(filemtime($name)), 11, '0', STR_PAD_LEFT) . "\0" //time
    83. 

  83.         . '        ' //checksum (8 spaces)
    84. 

  84.         . ($is_dir ? '5' : '0') //typeflag
    85. 

  85.         . str_repeat("\0", 100) //linkname
    86. 

  86.         . 'ustar  ' //magic
    87. 

  87.         /*
    88. 

  88.          * version (1) + username (32) + groupname (32) + devmajor (8) +
    89. 

  89.          * devminor (8) + prefix (155) + end (12) = 248
    90. 

  90.          */
    91. 

  91.         . str_repeat("\0", 248);
    92. 

  92.         
    93. 

  93.         $checksum = str_pad(decoct(self::checksum($header)), 6, '0', STR_PAD_LEFT) . "\0 ";
    94. 

  94.         return substr_replace($header, $checksum, 148, strlen($checksum));
    95. 

  95.     }
    96.

    97. 

  97.     /**
    98. 

  98.      * @param DirectoryList $filenames List of files to add to the archive
    99. 

  99.      * @param string $prepend_path Added in the filepath inside the tar archive
    100. 

  100.      * @param int $base_dir_length Length of directory path to cut off from start
    101. 

  101.      */
    102. 

  102.     public function __construct(DirectoryList $filenames, $prepend_path = '', $base_dir_length = 0)
    103. 

  103.     {
    104. 

  104.         $this -> base_dir_length = (int)$base_dir_length;
    105. 

  105.         $this -> prepend_path = Item::make_sure_slash($prepend_path);
    106. 

  106.         foreach ($filenames as $base)
    107. 

  107.         {
    108. 

  108.             $name = $filenames -> __get('dir_name') . $base;
    109. 

  109.             if (@is_dir($name))
    110. 

  110.             {
    111. 

  111.                 if ($base != '.' && $base != '..')
    112. 

  112.                 {
    113. 

  113.                     echo $this -> create_header($name);
    114. 

  114.                     $list = new DirectoryList($name);
    115. 

  115.                     new Tar($list, $this -> prepend_path, $this -> base_dir_length);
    116. 

  116.                 }
    117. 

  117.             }
    118. 

  118.             else if (@is_file($name) && @is_readable($name) && ($size = @filesize($name)))
    119. 

  119.             {
    120. 

  120.                 echo $this -> create_header($name, $size, false);
    121. 

  121.                 Url::force_download($name, false);
    122. 

  122.                 echo str_repeat("\0", (ceil($size / 512) * 512) - $size);
    123. 

  123.             }
    124. 

  124.         }
    125. 

  125.     }
    126. 

  126. }
