    72.

  74.                 $uploaded_files .= "<li>$filename</li>";

    76. 

  76.             }
    77. 

  77.             else
    78. 

  78.             {

  79.                 $errors .= "<li>$filename</li>";

  81.         }

  85.         }

  86.         if ($uploaded_files == '')

  87.         {
    88.

  89.         }

  91.         . '<strong>' . $words -> __get('uploaded files')

  102.     }
